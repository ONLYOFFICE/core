//
// глобальные константы
//

sampler BaseTex; // текстура

float center = 0;
float radius = 0;
float degree = 0;

//
// стуктуры
//

struct PS_INPUT
{
     float2 base : TEXCOORD0;
};

struct PS_OUTPUT
{
     vector diffuse : COLOR0;
};

//
// главная функция
//

PS_OUTPUT Main(PS_INPUT input)
{
	// Обнуляем члены выходной структуры
	PS_OUTPUT output = (PS_OUTPUT)0;

	float x = input.base.x - center;
	
	float factor = x * radius;

	factor *= factor;
	
	if( factor < 1.0f )
	{
		factor = pow( factor, degree );
		
		x *= factor;	
	}

	input.base.x = x + center;

	
	output.diffuse = tex2D(BaseTex, input.base);
     
	return output;
}
